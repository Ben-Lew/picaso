import pandas as pd

#picaso
from picaso import justdoit as jdi

#plotting
from bokeh.io import output_notebook
output_notebook()

Connect to Opacity Database¶

[2]:

opacity = jdi.opannection()

Load blank slate¶

[3]:

mars = jdi.inputs()

Set Planet & Star Properties¶

[4]:

#phase angle
mars.phase_angle(0) #radians

#define gravity
mars.gravity(gravity=3.7, gravity_unit=u.Unit('m/(s**2)')) #any astropy units available


#define star
mars.star(opacity,5000,0,4.0) #pysynphot database, temp, metallicity, logg

[10]:

mars.atmosphere(df=pd.DataFrame({'pressure':np.logspace(-6,2,40),
                         'temperature':np.linspace(270,100,40),
                         'H2O':np.zeros(40)+0.01,
                         'CO2':np.zeros(40)+1-0.01}))

mars.surface_reflect(0.3,opacity.wno)

Note, doing this will show you a message that Raman scattering was turned off. The Raman scattreing cross sections in the code are only valid for H2 atmospheres. Therefore, if you haven’t manually turned it off, it will do so for you. If you hate the error message you can turn off raman scattering but doing the following:

[11]:

mars.approx(raman='none')

#now when you set your atmosphere, you shouldn't see the message
#mars.atmosphere(filename = 'path/to/file',delim_whitespace=True)
mars.atmosphere(df=pd.DataFrame({'pressure':np.logspace(-6,2,40),
                         'temperature':np.linspace(100,270,40),
                         'H2O':np.zeros(40)+0.01,
                         'CO2':np.zeros(40)+1-0.01}))
